def HuellaNutricional
    totalValorCalorico=self.indice_energia(self.ValorCaloricoTotal)
    totalValorAmbiental=self.indice_huellaCarbono(self.ValorAmbientalTotal)
    return ((totalValorCalorico+totalValorAmbiental)/2).round(2)
end

def ValorAmbientalTotal 
    return @alimentos.zip(@cant_gramos)
                    .map{|item| ((item[0].gei+item[0].terreno)*item[1])/100}
                    .reduce(:+)
end

def ValorTotalGases
    total_valorGases = 0
    head_alimentos=@alimentos.head
    head_gramos=@cant_gramos.head
    while(head_alimentos != nil && head_gramos != nil)
        total_valorGases += (head_alimentos.value.gei * head_gramos.value)/100
        head_alimentos=head_alimentos.next
        head_gramos=head_gramos.next
    end
    return total_valorGases.round(2)
end

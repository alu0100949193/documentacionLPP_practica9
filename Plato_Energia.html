def porcentaje_glucidos
    total_glucidos = 0
    head_alimentos=@alimentos.head
    head_gramos=@cant_gramos.head
    while(head_alimentos != nil && head_gramos != nil)
        total_glucidos += (head_alimentos.value.glucidos * head_gramos.value)/100
        head_alimentos=head_alimentos.next
        head_gramos=head_gramos.next
    end
    total_gramos_g=@cant_gramos.reduce(:+)
    porcentaje_c=((total_glucidos/total_gramos_g)*100).round(2)
end

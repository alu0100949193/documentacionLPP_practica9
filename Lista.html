def gei_diaria
    gei_diaria = 0

def pop_head
    if(self.is_empty? == false)
        if (@head == @tail)
            value = @head.value
            @head = nil
            @tail = nil
            return value
        else
            value = @head.value
            @head.next.prev = nil
            @head = @head.next
            return value
        end
    end
end

def to_s
    current = @head
    string = "["
    while(current != nil)
        if (current.next == nil)
            string += "#{current.value}"
        else
            string += "#{current.value},"
        end
        current = current.next
    end
    return string += "]"
end
